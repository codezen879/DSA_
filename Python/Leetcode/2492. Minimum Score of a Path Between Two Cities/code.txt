
class Solution:
    def minScore(self, n: int, roads: List[List[int]]) -> int:
        # Adjacency list
        graph = [[] for _ in range(n + 1)]

        # Build the graph
        for u, v, w in roads:
            graph[u].append((v, w))
            graph[v].append((u, w))

        visited = [False] * (n + 1)
        queue = deque([1])
        visited[1] = True

        ans = float('inf')

        while queue:
            node = queue.popleft()

            for neighbor, weight in graph[node]:
                # Consider every edge
                ans = min(ans, weight)

                # Visit each node only once
                if not visited[neighbor]:
                    visited[neighbor] = True
                    queue.append(neighbor)

        return ans
